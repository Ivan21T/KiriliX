Интуитивен синтаксис с български думи

KiriliX използва български ключови думи, които са лесни за запомняне и правят кода по-четим за българските програмисти.

  • Ключови думи като "функция", "ако", "за", "докато"
  • Типове данни: "цялоЧисло", "дробночисло", "текст", "булев"
  • Функции за вход/изход: "прочети", "изведи", "отвори", "затвори"
  • Съобщения за грешки на български език
български-пример.krx
// Изчисляване на факториел
функция факториел(число) {
  ако (число <= 1) {
    върни 1;
  }
  върни число * факториел(число - 1);
}
// Главна програма
функция главна() {
  изведи("Добре дошли в KiriliX!");
  нека число = 5;
  нека резултат = факториел(число);
  изведи("Факториел от " + число + " е " + резултат);
}
главна();
